Module Values.Row

Row oriented API.

Each row is a record of values (encoded to a json object). An example of use would be Row.(empty |> int "x" 1 |> float "y" 2.)

type t
val empty : t
val int : string -> int -> t -> t
val float : string -> float -> t -> t
val string : string -> string -> t -> t
val to_json : t -> json
